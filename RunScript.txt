# Load the image and name it
load input/khoury.ppm khoury
save input/khoury.ppm khoury

# Create a new image using the intensity component
intensity-component khoury khoury-intensity
save input/khoury-intensity.ppm khoury-intensity

# Flip the intensity image horizontally
horizontal-flip khoury-intensity khoury-intensity-horizontal
save input/khoury-intensity-horizontal.ppm khoury-intensity-horizontal

# Create a new image using the blue component
blue-component khoury khoury-blue
save input/khoury-blue.ppm khoury-blue

# Blur the blue component image
blur khoury-blue khoury-blue-blur
save input/khoury-blue-blur.ppm khoury-blue-blur

# Brighten the original image by 25
brighten 25 khoury khoury-brighter
save input/khoury-brighter.ppm khoury-brighter

# Add sepia filter to the brightened image
sepia khoury-brighter khoury-brighter-sepia
save input/khoury-brighter-sepia.ppm khoury-brighter-sepia

# Flip the sepia image vertically
vertical-flip khoury-brighter-sepia khoury-sepia-vertical
save input/khoury-sepia-vertical.ppm khoury-sepia-vertical

# Combine the RGB components back, using the brightened red channel
brighten 25 khoury-red khoury-red-brighter
rgb-combine khoury-red-tint khoury-red-brighter khoury-green khoury-blue
save input/khoury-red-tint.ppm khoury-red-tint
